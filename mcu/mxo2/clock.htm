程式如下： 

module main(clk, seg_out);
  
input clk;
output [8:0] seg_out;
reg [8:0] seg_def[9:0];
reg [3:0] seg_cnt;
reg [23:0] clk_cnt;
assign seg_out = seg_def[seg_cnt];

initial begin
  clk_cnt = 0;
  seg_cnt = 0;
  seg_def[0] = 9'h3f;
  seg_def[1] = 9'h06;
  seg_def[2] = 9'h5b;
  seg_def[3] = 9'h4f;
  seg_def[4] = 9'h66;
  seg_def[5] = 9'h6d;
  seg_def[6] = 9'h7d;
  seg_def[7] = 9'h07;
  seg_def[8] = 9'h7f;
  seg_def[9] = 9'h6f;
end

always @(posedge clk) begin
  clk_cnt = clk_cnt + 1;
  if (clk_cnt > 12000000) begin
    clk_cnt = 0;
    seg_cnt = seg_cnt + 1;
    if (seg_cnt > 9)
      seg_cnt = 0;
  end
end
endmodule

開發板上連接的是12MHz振盪器，因此，每數到12000000時，就更新七段顯示器的數值
