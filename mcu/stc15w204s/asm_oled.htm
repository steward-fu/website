main.s 

  i2c_sda set 0cch
  i2c_scl set 0cdh
 
  ssd1306_address             equ 0x78
  ssd1306_lcdwidth            equ 128
  ssd1306_lcdheight           equ 64
  ssd1306_setcontrast         equ 0x81
  ssd1306_displayallow_resume equ 0xa4
  ssd1306_displayallow        equ 0xa5
  ssd1306_normaldisplay       equ 0xa6
  ssd1306_invertdisplay       equ 0xa7
  ssd1306_displayoff          equ 0xae
  ssd1306_displayon           equ 0xaf
  ssd1306_setdisplayoffset    equ 0xd3
  ssd1306_setcompins          equ 0xda
  ssd1306_setvcomdetect       equ 0xdb
  ssd1306_setdisplayclockdiv  equ 0xd5
  ssd1306_setprecharge        equ 0xd9
  ssd1306_setmultiplex        equ 0xa8
  ssd1306_setlowcolumn        equ 0x00
  ssd1306_sethighcolumn       equ 0x10
  ssd1306_setstartline        equ 0x40
  ssd1306_memorymode          equ 0x20
  ssd1306_columnaddr          equ 0x21
  ssd1306_pageaddr            equ 0x22
  ssd1306_comscaninc          equ 0xc0
  ssd1306_comscandec          equ 0xc8
  ssd1306_segremap            equ 0xa0
  ssd1306_chargepump          equ 0x8d
  ssd1306_externalvcc         equ 0x01
  ssd1306_switchcapvcc        equ 0x02

  .org 0h
  jmp _start

_start:
  call ssd1306_init
  jmp $

delay:
  mov r0, #10
  djnz r0, $
  ret

i2c_start:
  setb i2c_sda
  setb i2c_scl
  call delay
  clr i2c_sda
  call delay
  clr i2c_scl
  ret

i2c_stop:
  call delay
  clr i2c_sda
  call delay
  setb i2c_scl
  call delay
  setb i2c_sda
  ret

i2c_write:
  push 4
  push 5
  mov r4, #8
  mov r5, a
i0:
  mov a, r5
  rlc a
  mov r5, a
  jc i1
  clr i2c_sda
  sjmp i2
i1:
  setb i2c_sda
i2:
  call delay
  setb i2c_scl
  call delay
  clr i2c_scl
  djnz r4, i0
  
  call delay
  setb i2c_sda
  call delay
  setb i2c_scl
  call delay
  mov c, i2c_sda
  call delay
  clr i2c_scl
  pop acc
  mov r5, a
  pop acc
  mov r4, a
  ret

ssd1306_cmd:
  push 4
  mov r4, a
  call i2c_start
  mov a, #ssd1306_address
  call i2c_write
  mov a, #0
  call i2c_write
  mov a, r4
  call i2c_write
  call i2c_stop
  pop acc
  mov r4, a
  ret

 ssd1306_data:
  push 4
  mov r4, a
  call i2c_start
  mov a, #ssd1306_address
  call i2c_write
  mov a, #40h
  call i2c_write
  mov a, r4
  call i2c_write
  call i2c_stop
  pop acc
  mov r4, a
  ret

ssd1306_init:
  mov a, #ssd1306_displayoff
  call ssd1306_cmd
  mov a, #ssd1306_setdisplayclockdiv
  call ssd1306_cmd
  mov a, #80h
  call ssd1306_cmd
  mov a, #ssd1306_setmultiplex
  call ssd1306_cmd
  mov a, #3fh
  call ssd1306_cmd
  mov a, #ssd1306_setdisplayoffset
  call ssd1306_cmd
  mov a, #0
  call ssd1306_cmd
  mov a, #ssd1306_setstartline
  call ssd1306_cmd
  mov a, #ssd1306_chargepump
  call ssd1306_cmd
  mov a, #14h
  call ssd1306_cmd
  mov a, #ssd1306_memorymode
  call ssd1306_cmd
  mov a, #0
  call ssd1306_cmd
  mov a, #ssd1306_segremap | 1
  call ssd1306_cmd
  mov a, #ssd1306_comscandec
  call ssd1306_cmd
  mov a, #ssd1306_setcompins
  call ssd1306_cmd
  mov a, #12h
  call ssd1306_cmd
  mov a, #ssd1306_setcontrast
  call ssd1306_cmd
  mov a, #0cfh
  call ssd1306_cmd
  mov a, #ssd1306_setprecharge
  call ssd1306_cmd
  mov a, #0f1h
  call ssd1306_cmd
  mov a, #ssd1306_setvcomdetect
  call ssd1306_cmd
  mov a, #40h
  call ssd1306_cmd
  mov a, #ssd1306_displayallow_resume
  call ssd1306_cmd
  mov a, #ssd1306_normaldisplay
  call ssd1306_cmd
  mov a, #ssd1306_displayon
  call ssd1306_cmd
  ret
  .end
