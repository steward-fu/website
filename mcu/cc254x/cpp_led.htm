main.c 

#include "../../../include/cc2540.h"

#define LED0 P1_0
#define LED1 P1_1
#define LED2 P1_2
#define LED3 P1_3

void delay(unsigned int cnt)
{ 
  unsigned int ms=0;
  while(cnt--){
    for(ms=0; ms<512; ms++);
  } 
}

void SetLED(unsigned char mode)
{   
  LED0 = mode;
  LED1 = mode;
  LED2 = mode;
  LED3 = mode;
}

void InitGPIO(void)
{
  P1DIR|= 0x0f;
  P1SEL&= ~0x0f;
  SetLED(0);
}

void main(void)
{
  unsigned char idx=0;

  InitGPIO();     
  while(1){
    idx^= 1;
    SetLED(idx);
    delay(1000);
  }
}

編譯 

$ sdcc main.c
$ packihx main.ihx > main.hex
