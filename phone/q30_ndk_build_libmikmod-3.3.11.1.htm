手機 - Blackberry Passport - Core Native (C/C++) - Build libmikmod 3.3.11.1


步驟如下： 

$ cd
$ wget https://sources.voidlinux.eu/libmikmod-3.3.11.1/libmikmod-3.3.11.1.tar.gz
$ tar xvf libmikmod-3.3.11.1.tar.gz
$ cd libmikmod-3.3.11.1
$ source ~/bbndk/bbndk-env_10_3_1_995.sh
$ export TARGETMACH=arm-unknown-nto-qnx8.0.0eabi
$ export BUILDMACH=i686-pc-linux-gnu
$ export CROSS=arm-unknown-nto-qnx8.0.0eabi
$ export CC=${CROSS}-gcc
$ export LD=${CROSS}-ld
$ export AS=${CROSS}-as
$ export CXX=${CROSS}-g++
$ ./configure --host=$TARGETMACH --enable-static --prefix=$QNX_TARGET
$ make 
$ make install
