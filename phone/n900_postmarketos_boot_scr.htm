手機 - Nokia N900 - PostMarketOS - Boot.scr


參考資訊：
https://wiki.postmarketos.org/wiki/Nokia_N900

uboot-script.cmd 

setenv mmcnum 0
setenv mmcpart 1
setenv mmctype fat 
setenv setup_omap_atag 1
setenv bootargs init=/init.sh rw console=tty0 console=tty02 PMOS_NO_OUTPUT_REDIRECT
setenv mmckernfile /uImage-postmarketos-stable
setenv mmcinitrdfile /uInitrd-postmarketos-stable
setenv mmcscriptfile
echo Loading initramfs
run initrdload
echo Loading kernel
run kernload
echo Booting kernel
run kerninitrdboot

編譯 

$ mkimage -A arm -O linux -T script -C none -a 0 -e 0 -n postmarketos -d uboot-script.cmd boot.scr
